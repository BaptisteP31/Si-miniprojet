Mise en oeuvre du module NodeMCU Lolin ESP8266

Lobjectif est de mettre en oeuvre le module NodeMCU Lolin ESP 8266.

LESP8266

LESP8266 est un circuit intgr microcontrleur (32-bit RISC CPU Tensilica Xtensa LX106 80 MHz) avec connexion WiFi dvelopp par le fabricant chinois Espressif.

LESP8266 peut se programmer de plusieurs faons :

  • avec des scripts Lua avec le firmware NodeMCU
  • en C++ avec lIDE Arduino (Linux)
  • en C avec le SDK dEspressif
  • en C avec le SDK esp-open-sdk bas sur la chane de compilation GCC

Remarques : il existe aussi des possibilits avec Javascript et Python.

Flasher le module

Pour flasher le firmware, il faut installer esptool :

$ git clone https://github.com/themadinventor/esptool.git

$ cd esptool
$ python setup.py install
$ esptool.py --port /dev/ttyUSB0 write_flash --flash_mode qio --flash_size 4MB 0x00000 [firmware].bin

// ou :
$ python esptool.py --port /dev/ttyUSB0 write_flash --flash_mode qio --flash_size 4MB 0x00000 [firmware].bin

Firmware :

  • Commandes AT : at-v1.5.4.bin
  • LUA : nodemcu-float-0.9.6-dev_20150704.bin

Exemple de communication :

  1. Client TCP

Prparation du module :

AT+CWMODE=3
AT+CWJAP="Beauregard","password"
AT+CIPSTA?
+CIPSTA:ip:"192.168.52.27"
+CIPSTA:gateway:"192.168.52.1"
+CIPSTA:netmask:"255.255.255.0"

Dmarrer avec netcat un serveur TCP sur le port 5000 sur une machine GNU/Linux :

$ nc -l -p 5000

Ouvrir une connexion TCP partir du module et envoyer des donnes (CRLF la fin) :

AT+CIPMUX=0
AT+CIPSTART="TCP","192.168.52.2",5000
AT+CIPSEND=4
>ok

Remarques :

  • Si le module reoit des donnes, on aura une trame AT +IPD,n:xxxxxxxxxx avec n pour le nombre doctets reus et xxxxxxxxxx les donnes.
  • La commande AT+CIPSTART peut aussi tre utilise pour le protocole UDP.

Fermer la communication :

AT+CIPCLOSE
  1. Serveur TCP

Il faut activer les connexions multiples :

  • 0 : single connection
  • 1 : multiple connection
AT+CIPMUX=1

Puis dmarrer un serveur sur le port 5001 pour le module :

AT+CIPSERVER=1,5001

Utiliser un client netcat pour vous connecter au serveur :

$ nc 192.168.52.27 5001
blabla...

Et on peut aussi envoyer des donnes (CRLF la fin) :

AT+CIPSEND=0,4
> ok
  1. UART-WiFi passthrough

Le module ESP8266 peut fonctionner en UART-WiFi passthrough. Ce mode ne peut tre activ que lorsque lESP8266 fonctionne en tant que client TCP ou en UDP. On configure ce mode avec la commande AT+CIPMODE.

Lien : esp8266_at_command_examples_en.pdf

  1. Test WiFi :
#include "ESP8266WiFi.h"

void setup() 
{
  Serial.begin(115200);  
  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(2000);
}

void loop() {

  int n = WiFi.scanNetworks();

  if (n > 0)
  {
    for (int i = 0; i < n; ++i)
    {
      Serial.print(i + 1);
      Serial.print(": ");
      Serial.print(WiFi.SSID(i));
      Serial.print(" (");
      Serial.print(WiFi.RSSI(i));
      Serial.print(")");
      Serial.println((WiFi.encryptionType(i) == ENC_TYPE_NONE)?" ":"*");
      delay(10);
    }
  }
  Serial.println("");

  delay(5000);
}

#include "Serveur.h"

Serveur serveur; //serveur(80);

Avec une classe Serveur :

#ifndef SERVEUR_H
#define SERVEUR_H

#include <ESP8266WiFi.h>
#include <WiFiServer.h>

static const char* ssid = "Beauregard";
static const char* password = "password";

class Serveur : public WiFiServer
{
  private:
    int ledPin2; // Led bleue
    int etatLed;
    
    void demarrerWiFi();
    void demarrerServeur();
    void envoyerPageAccueil(WiFiClient &client);
  
  public:
    Serveur(uint16_t port=80);
    virtual ~Serveur() {}
    
    void demarrer();
    void traiterRequete();
};

#endif
#include "Serveur.h"

Serveur::Serveur(uint16_t port/*=80*/) : WiFiServer(port)
{
  ledPin2 = 02; // Led bleue  
  pinMode(ledPin2, OUTPUT);
  etatLed = LOW;
  digitalWrite(ledPin2, etatLed);
}

void Serveur::demarrer()
{
  demarrerWiFi();
  demarrerServeur();
}

void Serveur::traiterRequete()
{
  WiFiClient client = available();
  if (!client) 
  {
    return;
  }
  
  Serial.print("Requte client ");
  while(!client.available())
  {
    delay(1);
  }
  Serial.println(client.remoteIP());  
  
  //String requete = client.readStringUntil('\r');  
  //Serial.println(requete);
  //client.flush();

  String requete;
  while(client.available())
  {
    String ligne = client.readStringUntil('\r');
    Serial.print(ligne);
    requete += ligne;
  }
  client.flush();

  if(requete.indexOf("GET") != -1)
  {
    envoyerPageAccueil(client);    
  }
  else if(requete.indexOf("POST") != -1)
  {
      if(requete.indexOf("led") != -1)
      {          
        if(requete.indexOf("led=0") != -1)
        {
          etatLed = HIGH;
          digitalWrite(ledPin2, etatLed);  
        }
        else if(requete.indexOf("led=1") != -1)
        {
          etatLed = LOW;
          digitalWrite(ledPin2, etatLed);
        }
        envoyerPageAccueil(client);    
      }
  }
  else 
  {
    Serial.println("Requte invalide !");
    client.stop();
    return;
  }
  
  Serial.println("");
}

void Serveur::demarrerWiFi()
{
  Serial.println("");
  Serial.println("");
  Serial.print("Connexion au rseau WiFi ");
  Serial.println(ssid);
  
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  
  while (WiFi.status() != WL_CONNECTED) 
  {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.print("Adresse IP du serveur : ");
  Serial.println(WiFi.localIP());  
}

void Serveur::demarrerServeur()
{  
  begin();
  Serial.println("Dmarrage du serveur Web");
  Serial.println("");
}

void Serveur::envoyerPageAccueil(WiFiClient &client)
{  
  String reponse = "HTTP/1.1 200 OK\r\nContent-Type: text/html\r\nPragma: no-cache\r\n\r\n<!DOCTYPE HTML>\r\n";
  reponse += "<html>\n";
  reponse += "<head>\n";
  reponse += "<title>Serveur ESP8266</title>\n";
  reponse += "</head>\n";
  reponse += "<body>\n";
  reponse += "<h1>Bienvenue sur le serveur ESP8266 WiFi !</h1>\n";
  reponse += "<p>\n";
  reponse += "<form method=\"POST\">\n";  
  if (etatLed == LOW) 
  {
    reponse += "<p><input type=\"radio\" id=\"led\" name=\"led\" value=\"0\" />LED off\n";
    reponse += "<br /><input type=\"radio\" id=\"led\" name=\"led\" value=\"1\" checked />LED on</p>";
    
  } 
  else 
  {
    reponse += "<p><input type=\"radio\" id=\"led\" name=\"led\" value=\"0\" checked>LED off</input>\n";
    reponse += "<br /><input type=\"radio\" id=\"led\" name=\"led\" value=\"1\">LED on</input></p>";
  }
  reponse += "<p><input type=\"submit\" /></p>\n";
  reponse += "</form>\n";
  reponse += "</p>\n";
  reponse += "</body>\n";
  reponse += "</html>\n";

  client.flush();
  client.print(reponse);
  //delay(1);  
}
